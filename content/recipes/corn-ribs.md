dietaryTags: "Veg, Gluten-Free, Dairy-Free, Seasonal"
ingredients:
  - name: "Corn Cobs"
    amount: "2 medium"
  - name: "Korean bbq seasoning"
    amount: "1 packet"
  - name: "Olive oil"
    amount: "1/4 cup"
  - name: "Bbq sauce"
    amount: "1/2 cup"
  - name: "Parmesan Cheese (freshly grated)"
    amount: "1/4 cup"
  - name: "Cilantro (chopped)"
    amount: "1/4 cup"
steps:
  - title: "Prepare the Corn"
    body: "With a sharp knife, carefully stand each cob of corn upright and cut into 4 pieces. Place the corn pieces in a large bowl, drizzle with olive oil, and season generously with Korean BBQ seasoning. Toss well to coat and let sit for a few minutes."
  - title: "Cook"
    body: "Grill the corn on the BBQ until slightly browned and the edges start to curl up."
  - title: "Serve"
    body: "Return the grilled corn to the same bowl, add BBQ sauce, and toss until evenly coated. Transfer to a serving dish, then finish with freshly grated Parmesan and chopped cilantro. Serve immediately."
howToServe:
  - "Best served fresh."
  - "Serve alongside grilled proteins like fish, chicken, or tofu for a smoky summer meal."
  - "Make them the centerpiece of a summer spread with salads, dips, and fresh bread."
chefsNote: "Fresh corn works best. Swap Parmesan cheese for vegan cheese if Vegan."
