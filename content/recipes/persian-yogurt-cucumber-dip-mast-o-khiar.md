emoji: "🍽️"
dietaryTags: "Veg, Gluten-Free, Contains Nuts"
ingredients:
  - name: "Thick yogurt (Skyr or Greek)"
    amount: "3 cups"
  - name: "Persian cucumbers"
    amount: "4 medium"
  - name: "Fresh cilantro"
    amount: "½ cup"
  - name: "Fresh mint"
    amount: "6–7 leaves"
  - name: "Walnuts"
    amount: "½ cup"
  - name: "Raisins"
    amount: "½ cup"
  - name: "Lemon juice"
    amount: "1 tbsp"
  - name: "Salt and pepper"
    amount: "To taste"
steps:
  - title: ""
    body: "Grate the cucumbers. Using a cheesecloth, squeeze out excess liquid."
  - title: ""
    body: "Finely chop the cilantro and slice the mint into thin strips. Chop the walnuts."
  - title: ""
    body: "In a bowl, combine yogurt, cucumber, herbs, walnuts, and raisins."
  - title: ""
    body: "Add lemon juice, salt, and pepper. Mix well and adjust to taste."
  - title: ""
    body: "Transfer to a serving bowl and garnish with dried rose petals or fresh mint, if desired."
howToServe:
  - "Serve chilled as a dip or alongside grilled meats, vegetables, or flatbread."
  - "Can be prepared ahead of time, but add the raisins closer to serving so they don’t soften too much."
chefsNote: "Best served fresh, but can be stored in the fridge for up to 2 days (without the raisins). Remove walnuts if nut-free."
