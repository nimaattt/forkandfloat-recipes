emoji: "🍽️"
dietaryTags: "Veg, Dairy-Free"
ingredients:
  - name: "Shallot"
    amount: "1 Medium"
  - name: "Garlic"
    amount: "3 cloves"
  - name: "Spinach"
    amount: "3 cups"
  - name: "Ricotta Cheese"
    amount: "1 cup"
  - name: "Zucchini"
    amount: "1 small"
  - name: "Jumbo Shell Pasta"
    amount: "1 bag"
  - name: "Salt and Pepper"
    amount: "To taste"
  - name: "Butter"
    amount: "1/4 cup"
  - name: "All purpose flour"
    amount: "1 tbsp"
  - name: "Milk of choice"
    amount: "1 cup"
  - name: "Lemon Zest"
    amount: "(zest of )1 lemon"
  - name: "Freshly grated Parmesan Cheese"
    amount: "1/2 cup"
  - name: "Olive Oil"
    amount: "1 tbsp"
steps:
  - title: "Sauté the base"
    body: "Heat a pan over medium heat with olive oil. Add finely chopped shallot and garlic and let them gently sizzle until soft and fragrant."
  - title: "Cook the spinach"
    body: "Add the spinach (yes, it will feel like too much). Cook, stirring, until completely wilted and the moisture has cooked off. Remove from heat."
  - title: "Make the filling"
    body: "Add ricotta to the spinach mixture. Season with salt and pepper, stir to combine, and set aside."
  - title: "Prepare the béchamel"
    body: "In a saucepan, melt butter over medium heat. Add flour and cook for about 1 minute, stirring constantly. Slowly pour in milk (or cream), whisking to avoid lumps."
  - title: "Season"
    body: "Season the sauce with salt and pepper, then add lemon zest. Continue stirring until the béchamel is thick, smooth, and glossy. Remove from heat."
  - title: "Cook the pasta"
    body: "Bring a large pot of well-salted water to a boil. Cook jumbo shells until al dente. Drain and let cool slightly for easier handling."
  - title: "Fill the shells"
    body: "Spoon the ricotta-spinach mixture into each shell."
  - title: "Assemble"
    body: "Arrange the filled shells in an oven-safe dish. Pour béchamel generously over the top."
  - title: "Finish & bake"
    body: "Grate zucchini evenly over the dish for a subtle crisp texture. Finish with a generous layer of parmesan. Bake at 350°F (175°C) for 20 minutes, until melted, bubbling, and beautifully cohesive."
howToServe:
  - "Serve warm, directly from the baking dish"
  - "Finish with a light drizzle of olive oil and a final crack of black pepper just before serving"
chefsNote: "This dish holds heat well and can sit 15–20 minutes before serving without losing texture For events, it can be fully baked ahead, then reheated uncovered at 375°F until bubbling and lightly golden"
