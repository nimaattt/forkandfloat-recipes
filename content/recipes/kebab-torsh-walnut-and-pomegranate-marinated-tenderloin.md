ingredients:
  - name: "Tenderloin"
    amount: "2 kg"
  - name: "Walnuts"
    amount: "1 cup"
  - name: "Onion"
    amount: "1 large"
  - name: "Garlic"
    amount: "3–4 cloves"
  - name: "Parsley"
    amount: "½ cup, chopped"
  - name: "Mint"
    amount: "½ cup, chopped"
  - name: "Olive oil"
    amount: "½ cup"
  - name: "Pomegranate molasses"
    amount: "1 cup"
  - name: "Pomegranate seeds"
    amount: "¾ cup"
  - name: "Lemon juice"
    amount: "Juice of ½ lemon"
  - name: "Black pepper"
    amount: "optional"
steps:
  - title: ""
    body: "Cut the tenderloin into roughly 2-inch cubes."
  - title: ""
    body: "In a blender, combine onion, garlic, parsley, mint, walnuts, olive oil, pomegranate molasses, and lemon juice. Blend until you get a thick paste-like marinade."
  - title: ""
    body: "Add the marinade to the meat and mix well until fully coated. Cover and refrigerate overnight for best flavor."
  - title: ""
    body: "Add the pomegranate seeds and mix again."
  - title: ""
    body: "Grill over charcoal barbecue or gas barbecue for about 4 minutes per side until slightly browned and cooked to your preferred doneness."
howToServe:
  - "Serve fresh with rice, grilled tomatoes, or vegetables on the side."
chefsNote: "Best served fresh, but can be stored in the fridge (uncooked) for up to 2 days. Charcoal grilling gives the kabob a more traditional smoky flavor, but gas barbecue works well too."
