ingredients:
  - name: "Kohlrabi"
    amount: "1 large head"
  - name: "All purpose flour"
    amount: "1 cup"
  - name: "Egg"
    amount: "1 large"
  - name: "Bread crumbs"
    amount: "1 cup"
  - name: "Garlic powder"
    amount: "1 tsp"
  - name: "Salt & Pepper"
    amount: "To taste"
  - name: "Neutral Oil"
    amount: "Enough for frying"
  - name: "Pickles"
    amount: "1 large"
  - name: "Kewpie mayo"
    amount: "1 tbsp"
  - name: "Ketchup"
    amount: "1 tbsp"
  - name: "Mustard"
    amount: "1 tbsp"
  - name: "Fresh dill"
    amount: "1 few leaves (for garnish)"
steps:
  - title: ""
    body: "Peel the kohlrabi and cut into about 1-inch thick slices."
  - title: ""
    body: "Add to salted boiling water and cook for 5–10 minutes until knife tender. Remove and pat dry."
  - title: ""
    body: "Prepare 3 separate bowls:"
  - title: ""
    body: "Flour mixed with salt, pepper, and garlic powder"
  - title: ""
    body: "Beaten egg"
  - title: ""
    body: "Bread crumbs"
  - title: ""
    body: "Coat each kohlrabi slice in flour, then egg, then bread crumbs. Repeat once more for a thicker and crunchier coating if desired."
  - title: ""
    body: "Heat oil in a pan and fry the schnitzels for about 5 minutes per side until golden brown and crispy."
  - title: ""
    body: "Finely chop the pickles and mix with kewpie mayo, ketchup, and mustard to make the sauce."
  - title: ""
    body: "Serve the schnitzels warm with the sauce and fresh dill on top."
howToServe:
  - "Can be served in smaller portions as mid day snack or as the main."
  - "Pairs well with fresh herbs, a simple salad, or roasted potatoes. Garnish with fresh dill for extra freshness and flavour."
  - "Serve hot for best result and crunch."
chefsNote: "Double coating creates a crunchier texture. Best served fresh and hot. Can skip egg if vegan."
