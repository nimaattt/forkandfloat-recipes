ingredients:
  - name: "Quince"
    amount: "4 large"
  - name: "Sugar"
    amount: "3 cups"
  - name: "Quince seeds"
    amount: "4–5 pieces"
  - name: "Lemon juice"
    amount: "1 tsp"
  - name: "Water"
    amount: "~10 cups"
steps:
  - title: ""
    body: "Cut quinces in half, then slice and cut into your desired size (about 1–1.5 inch pieces)."
  - title: ""
    body: "In a large pot, bring about 10 cups of water to a boil. Add sugar and stir until fully dissolved."
  - title: ""
    body: "Add the quince pieces and quince seeds to the pot. Bring to a boil on high heat."
  - title: ""
    body: "Reduce heat to medium and let cook for about 1 hour. Add lemon juice."
  - title: ""
    body: "Continue cooking for another 2–3 hours until the syrup thickens slightly and the colour deepens to a rich reddish tone."
  - title: ""
    body: "Turn off heat. Place a clean cloth under the lid, cover, and let it rest at room temperature for at least 8 hours."
howToServe:
  - "Served fully cooled down."
  - "Serve with some butter or cream and freshly toasted bread for breakfast."
  - "Serve on yogurt or chia pudding for a sweet and healthy breakfast or midday snack."
chefsNote: "Quince seeds help naturally thicken the jam. Remove before serving. Store in the fridge for up to 3 months."
